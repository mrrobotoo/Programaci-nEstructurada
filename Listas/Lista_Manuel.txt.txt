public class Total {

        // Agregar jugadores a la lista
        jugadores.add("Jugador1");

        jugadores.add("Jugador3");

        // Contador para el número de elementos en la lista
        int contador = 0;

public class Verificar {
    public static void main(String[] args) {
        List<String> jugadores = new ArrayList<>();
        jugadores.add("Jugador1");
        jugadores.add("Jugador2");
        jugadores.add("Jugador3");

        // Variable para almacenar el resultado
        boolean encontrado = false;

        // Usar un bucle for para verificar si "Jugador3" está en la lista
        for (int i = 0; i < jugadores.size(); i++) {
            if (jugadores.get(i).equals("Jugador3")) {
                encontrado = true;
                break; // Salir del bucle una vez encontrado
            }
        }

        // Imprimir el resultado
        System.out.println(encontrado); // Imprime "true"
    }
}

public class Eliminar {
    public static void main(String[] args) {
        List<String> jugadores = new ArrayList<>();
        jugadores.add("Jugador1");
        jugadores.add("Jugador2");
        jugadores.add("Jugador3");

        // Usar un bucle for para eliminar "Jugador2" de la lista
        for (int i = 0; i < jugadores.size(); i++) {
            if (jugadores.get(i).equals("Jugador2")) {
                jugadores.remove(i);
                break; // Salir del bucle una vez que se ha eliminado el elemento
            }
        }

        // Imprimir la lista de jugadores
        System.out.println(jugadores); // Imprime "[Jugador1, Jugador3]"
    }
}

public class Alfabéticamente {
    public static void main(String[] args) {
        // Crear la lista de jugadores
        List<String> jugadores = new ArrayList<>();

        // Agregar jugadores a la lista
        jugadores.add("b");
        jugadores.add("a");
        jugadores.add("c");

        // Implementar el algoritmo de burbuja para ordenar la lista
        for (int i = 0; i < jugadores.size() - 1; i++) {
            for (int j = 0; j < jugadores.size() - 1 - i; j++) {
                if (jugadores.get(j).compareTo(jugadores.get(j + 1)) > 0) {
                    // Intercambiar jugadores[j] y jugadores[j + 1]
                    String temp = jugadores.get(j);
                    jugadores.set(j, jugadores.get(j + 1));
                    jugadores.set(j + 1, temp);
                }
            }
        }

        // Imprimir la lista de jugadores ordenada
        System.out.println(jugadores); // Imprime "[Jugador1, Jugador2, Jugador3]"
    }
}
